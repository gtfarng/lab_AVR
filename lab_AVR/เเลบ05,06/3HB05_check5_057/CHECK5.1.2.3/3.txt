#include <avr/io.h>
#include <util/delay.h>
#define F_CPU 16000000UL

int main (void)
{	
	unsigned char switch_v ,count,i;
	unsigned char LOOKUPTB[]={  	0b00111111, 0b00000110,
				    	0b01011011, 0b01001111,
					0b01100110, 0b01101101,
					0b01111101, 0b00000111,
					0b01111111, 0b01101111,	
					0b01110111, 0b01111100,	
					0b00111001, 0b01011110,	
					0b01111001, 0b01110001 };

	DDRC = 0xFF;
	DDRB = 0xFF;
	DDRD = 0xFE;
	
	count = 0;
	PORTB = 0b01011100; // o
	PORTC = 0b01110001; // F
	

	    while(1)
		{
		switch_v = PIND; // input from PIND
		switch_v &= 0x01; // Check Bit equal 0
				
		if(switch_v == 0x00)
			{  		if(count == 0)
						{
			    		 PORTB = 0b01011100; // o
	 					 PORTC = 0b01010100; // n
						 PORTD = 0b00000010; //Lamp on
						 count = 1;
						 _delay_ms(8000); // delay 1 sec
						}

               		else if(count == 1)
						{

								for(i=0;i<21;i++)
								{
	    							_delay_ms(8000); // delay 1 sec		  
						 			PORTB = LOOKUPTB[(20-i)/10]; // Tens
	 	 			   	 			PORTC = LOOKUPTB[(20-i)%10]; // unit
									PORTD = ~PORTD;
									PORTD |= 0b00000010;
									PORTD &= 0b00000100;
										
										// # ข้างล่างนี้ทำผิดน่ะ 
										//if(i%2)
										//	PORTD = 0b00000100;
										//else
										//	PORTD = 0b00000000;
										//if(PORTD & 0b00000100 == 0x01)
										//	PORTD = 0b00000100;
										//else
										//	PORTD = 0b00000000;
								}

						PORTB = 0b01011100; // o
						PORTC = 0b01110001; // F
					    PORTD = 0b00000000; //Lamp off
						count = 0;
					
			
						}	
			  		
			}
		}
   



}
